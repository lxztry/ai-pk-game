🤖 创建你的第一个 Agent

from agents.code_agent import CodeAgent
from game.agent import Observation

class MyAgent(CodeAgent):
    def step(self, observation: Observation) -> str:
        # 发现敌人就射击
        if observation.enemies_in_view:
            return "shoot"
        # 否则向上移动
        return "move_up"
